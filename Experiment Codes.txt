#include <stdio.h>
#include <stdlib.h>

struct Node{
    int data;
    char color;
    struct Node *left,*right,*parent;
};

struct Node* root=NULL;

struct Node* newNode(int data){
    struct Node* n=(struct Node*)malloc(sizeof(struct Node));
    n->data=data;
    n->color='R';
    n->left=n->right=n->parent=NULL;
    return n;
}

void leftRotate(struct Node* x){
    struct Node* y=x->right;
    x->right=y->left;
    if(y->left) y->left->parent=x;
    y->parent=x->parent;
    if(!x->parent) root=y;
    else if(x==x->parent->left) x->parent->left=y;
    else x->parent->right=y;
    y->left=x;
    x->parent=y;
}

void rightRotate(struct Node* x){
    struct Node* y=x->left;
    x->left=y->right;
    if(y->right) y->right->parent=x;
    y->parent=x->parent;
    if(!x->parent) root=y;
    else if(x==x->parent->left) x->parent->left=y;
    else x->parent->right=y;
    y->right=x;
    x->parent=y;
}

void fixInsert(struct Node* pt){

    while(pt != root && pt->parent && pt->parent->color == 'R'){

        struct Node* v = pt->parent;

        if(!v->parent) break;   // Safety check

        struct Node* u = v->parent;

        if(v == u->left){

            struct Node* w = u->right;

            if(w && w->color == 'R'){
                v->color = 'B';
                w->color = 'B';
                u->color = 'R';
                pt = u;
            }
            else{

                if(pt == v->right){
                    pt = v;
                    leftRotate(pt);
                }

                if(pt->parent){
                    pt->parent->color = 'B';
                    if(pt->parent->parent){
                        pt->parent->parent->color = 'R';
                        rightRotate(pt->parent->parent);
                    }
                }
            }
        }
        else{

            struct Node* w = u->left;

            if(w && w->color == 'R'){
                v->color = 'B';
                w->color = 'B';
                u->color = 'R';
                pt = u;
            }
            else{

                if(pt == v->left){
                    pt = v;
                    rightRotate(pt);
                }

                if(pt->parent){
                    pt->parent->color = 'B';
                    if(pt->parent->parent){
                        pt->parent->parent->color = 'R';
                        leftRotate(pt->parent->parent);
                    }
                }
            }
        }
    }

    if(root) root->color = 'B';
}


void insert(int data){
    struct Node* n=newNode(data);
    struct Node *y=NULL,*x=root;
    while(x){
        y=x;
        if(n->data<x->data) x=x->left;
        else x=x->right;
    }
    n->parent=y;
    if(!y) root=n;
    else if(n->data<y->data) y->left=n;
    else y->right=n;
    fixInsert(n);
}

void levelOrder(){
    if(!root) return;
    struct Node* q[1000];
    int f=0,r=0;
    q[r++]=root;
    while(f<r){
        int c=r-f;
        while(c--){
            struct Node* t=q[f++];
            printf("%d(%c) ",t->data,t->color);
            if(t->left) q[r++]=t->left;
            if(t->right) q[r++]=t->right;
        }
        printf("\n");
    }
}

int main(){
    int n,x;
    char op;
    scanf("%d",&n);
    while(n--){
        scanf(" %c %d",&op,&x);
        if(op=='I') insert(x);
    }
    levelOrder();
    return 0;
}


Exp 2: fractional knapsack

#include <stdio.h>
#include <stdlib.h>

struct Item {
    int value;
    int weight;
    double ratio;
};


int compare(const void *a, const void *b) {
    struct Item *i1 = (struct Item *)a;
    struct Item *i2 = (struct Item *)b;

    if (i2->ratio > i1->ratio)
        return 1;
    else if (i2->ratio < i1->ratio)
        return -1;
    else
        return 0;
}

int main() {
    int n, W;
    scanf("%d", &n);
    scanf("%d", &W);

    struct Item items[n];

    for (int i = 0; i < n; i++) {
        scanf("%d %d", &items[i].value, &items[i].weight);
        items[i].ratio = (double)items[i].value / items[i].weight;
    }

    qsort(items, n, sizeof(struct Item), compare);

    double totalValue = 0.0;
    int remainingCapacity = W;

    for (int i = 0; i < n; i++) {
        if (remainingCapacity == 0)
            break;

        if (items[i].weight <= remainingCapacity) {
            totalValue += items[i].value;
            remainingCapacity -= items[i].weight;
        } else {
            totalValue += items[i].ratio * remainingCapacity;
            remainingCapacity = 0;
        }
    }

    printf("%.2f", totalValue);

    return 0;
}

Exp 3: job squencing

#include <stdio.h>

struct Job
{
    int id;
    int deadline;
    int profit;
};

int main()
{
    int n;
    scanf("%d", &n);

    struct Job jobs[100];

    int maxDeadline = 0;

    for (int i = 0; i < n; i++)
    {
        scanf("%d %d %d", &jobs[i].id, &jobs[i].deadline, &jobs[i].profit);

        if (jobs[i].deadline > maxDeadline)
            maxDeadline = jobs[i].deadline;
    }

    // Sort jobs by profit descending (Bubble sort)
    for (int i = 0; i < n - 1; i++)
    {
        for (int j = 0; j < n - i - 1; j++)
        {
            if (jobs[j].profit < jobs[j + 1].profit)
            {
                struct Job temp = jobs[j];
                jobs[j] = jobs[j + 1];
                jobs[j + 1] = temp;
            }
        }
    }

    int slot[101];
    for (int i = 0; i <= maxDeadline; i++)
        slot[i] = -1;

    int totalProfit = 0;

    // Job sequencing
    for (int i = 0; i < n; i++)
    {
        for (int j = jobs[i].deadline; j >= 1; j--)
        {
            if (slot[j] == -1)
            {
                slot[j] = jobs[i].id;
                totalProfit += jobs[i].profit;
                break;
            }
        }
    }

    // Output
    printf("Scheduled tasks: ");
    for (int i = 1; i <= maxDeadline; i++)
    {
        if (slot[i] != -1)
            printf("%d ", slot[i]);
    }

    printf("\nMaximum Profit: %d", totalProfit);

    return 0;
}

Exp 4:currency exchange

#include <stdio.h>
#include <limits.h>

int minCoins(int coins[], int n, int T) {
    int dp[T + 1];

    // Initialize dp array
    for (int i = 0; i <= T; i++) {
        dp[i] = INT_MAX;
    }

    dp[0] = 0; // 0 coins needed for amount 0

    // Fill dp array
    for (int i = 1; i <= T; i++) {
        for (int j = 0; j < n; j++) {
            if (coins[j] <= i && dp[i - coins[j]] != INT_MAX) {
                if (dp[i] > dp[i - coins[j]] + 1) {
                    dp[i] = dp[i - coins[j]] + 1;
                }
            }
        }
    }

    if (dp[T] == INT_MAX)
        return -1;
    else
        return dp[T];
}

int main() {
    int n, T;

    scanf("%d", &n);

    int coins[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &coins[i]);
    }

    scanf("%d", &T);

    int result = minCoins(coins, n, T);

    printf("%d\n", result);

    return 0;
}


Exp 5: matrix chain multiplication

#include <stdio.h>
#include <limits.h>

int main() {
    int n;
    scanf("%d", &n);

    int p[n+1];
    for (int i = 0; i <= n; i++) {
        scanf("%d", &p[i]);
    }

    int m[n][n];

    // cost is 0 when multiplying one matrix
    for (int i = 0; i < n; i++) {
        m[i][i] = 0;
    }

    // chain length
    for (int L = 2; L <= n; L++) {
        for (int i = 0; i < n - L + 1; i++) {
            int j = i + L - 1;
            m[i][j] = INT_MAX;

            for (int k = i; k < j; k++) {
                int cost = m[i][k] + m[k+1][j] + p[i]*p[k+1]*p[j+1];

                if (cost < m[i][j]) {
                    m[i][j] = cost;
                }
            }
        }
    }

    printf("%d\n", m[0][n-1]);

    return 0;
}

Exp 6: Rabin Karp

#include <stdio.h>
#include <string.h>

#define d 256  // number of characters in input alphabet

int main() {
    char text[1000], pattern[100];
    int i, j, t = 0, p = 0, h = 1;
    int n, m;
    int q = 101; // prime number

    // Input
    scanf("%s", text);
    scanf("%s", pattern);

    n = strlen(text);
    m = strlen(pattern);

    // Calculate h = pow(d, m-1) % q
    for (i = 0; i < m - 1; i++)
        h = (h * d) % q;

    // Calculate initial hash values
    for (i = 0; i < m; i++) {
        p = (d * p + pattern[i]) % q;
        t = (d * t + text[i]) % q;
    }

    int found = 0;

    // Slide pattern over text
    for (i = 0; i <= n - m; i++) {

        // Check hash values
        if (p == t) {
            // Check characters one by one
            for (j = 0; j < m; j++) {
                if (text[i + j] != pattern[j])
                    break;
            }

            if (j == m) {
                printf("Pattern found at index: %d\n", i);
                found = 1;
            }
        }

        // Calculate next window hash
        if (i < n - m) {
            t = (d * (t - text[i] * h) + text[i + m]) % q;

            // Convert negative to positive
            if (t < 0)
                t = t + q;
        }
    }

    // If not found
    if (!found) {
        printf("Pattern not found\n");
    }

    return 0;
}

Exp 7: LCS

#include <stdio.h>
#include <string.h>

int max(int a, int b)
{
    if(a > b)
        return a;
    else
        return b;
}

int main()
{
    char X[100], Y[100];
    char lcs[100];

    scanf("%s", X);
    scanf("%s", Y);

    int m = strlen(X);
    int n = strlen(Y);

    int L[m+1][n+1];

    for(int i=0;i<=m;i++)
    {
        for(int j=0;j<=n;j++)
        {
            if(i==0 || j==0)
                L[i][j]=0;
            else if(X[i-1]==Y[j-1])
                L[i][j]=L[i-1][j-1]+1;
            else
                L[i][j]=max(L[i-1][j],L[i][j-1]);
        }
    }

    printf("Length of LCS: %d\n",L[m][n]);

    int index=L[m][n];
    lcs[index]='\0';

    int i=m,j=n;

    while(i>0 && j>0)
    {
        if(X[i-1]==Y[j-1])
        {
            lcs[index-1]=X[i-1];
            i--;
            j--;
            index--;
        }
        else if(L[i][j-1] > L[i-1][j])
            j--;
        else
            i--;
    }

    printf("LCS: %s",lcs);

    return 0;
}


Exp 8: cea

#include <stdio.h>

// Recursive Function
int fibRec(int n)
{
    if(n==0)
        return 0;
    if(n==1)
        return 1;
    return fibRec(n-1) + fibRec(n-2);
}

// Non-Recursive Function
int fibIter(int n)
{
    int a=0,b=1,c,i;

    if(n==0)
        return 0;
    if(n==1)
        return 1;

    for(i=2;i<=n;i++)
    {
        c=a+b;
        a=b;
        b=c;
    }
    return c;
}

int main()
{
    int n;

    scanf("%d",&n);

    printf("Recursive Fibonacci(%d): %d\n",n,fibRec(n));
    printf("Non-Recursive Fibonacci(%d): %d",n,fibIter(n));

    return 0;
}
