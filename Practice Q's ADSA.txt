#include <stdio.h>

void hanoi(int n, char A, char B, char C) {
    if (n == 1) {
        printf("Move disk 1 from %c to %c\n", A, C);
        return;
    }
    hanoi(n - 1, A, C, B);
    printf("Move disk %d from %c to %c\n", n, A, C);
    hanoi(n - 1, B, A, C);
}

int main() {
    int n;
    scanf("%d", &n);
    hanoi(n, 'A', 'B', 'C');
    return 0;
}

2) Quick Sort

#include <stdio.h>

void quicksort(int a[], int l, int h) {
    if (l < h) {
        int i = l, j = h, pivot = a[l], temp;
        while (i < j) {
            while (a[i] <= pivot && i < h) i++;
            while (a[j] > pivot) j--;
            if (i < j) {
                temp = a[i]; a[i] = a[j]; a[j] = temp;
            }
        }
        a[l] = a[j];
        a[j] = pivot;

        quicksort(a, l, j - 1);
        quicksort(a, j + 1, h);
    }
}

int main() {
    int n, a[100], i;
    scanf("%d", &n);
    for (i = 0; i < n; i++) scanf("%d", &a[i]);

    quicksort(a, 0, n - 1);

    for (i = 0; i < n; i++) printf("%d ", a[i]);
    return 0;
}


3) 0/1 Knapsack

#include <stdio.h>

int max(int a, int b) { return (a > b) ? a : b; }

int main() {
    int n, W, i, j;
    int val[100], wt[100], dp[101][101];

    scanf("%d", &n);
    for (i = 0; i < n; i++) scanf("%d", &val[i]);
    for (i = 0; i < n; i++) scanf("%d", &wt[i]);
    scanf("%d", &W);

    for (i = 0; i <= n; i++) {
        for (j = 0; j <= W; j++) {
            if (i == 0 || j == 0)
                dp[i][j] = 0;
            else if (wt[i-1] <= j)
                dp[i][j] = max(val[i-1] + dp[i-1][j - wt[i-1]], dp[i-1][j]);
            else
                dp[i][j] = dp[i-1][j];
        }
    }

    printf("%d", dp[n][W]);
    return 0;
}

5) Merge Sort

#include <stdio.h>

void merge(int a[], int l, int m, int h) {
    int i = l, j = m + 1, k = l, b[100];

    while (i <= m && j <= h) {
        if (a[i] < a[j])
            b[k++] = a[i++];
        else
            b[k++] = a[j++];
    }

    while (i <= m) b[k++] = a[i++];
    while (j <= h) b[k++] = a[j++];

    for (i = l; i <= h; i++)
        a[i] = b[i];
}

void mergesort(int a[], int l, int h) {
    if (l < h) {
        int m = (l + h) / 2;

        mergesort(a, l, m);
        mergesort(a, m + 1, h);
        merge(a, l, m, h);
    }
}

int main() {
    int n, a[100], i;

    scanf("%d", &n);
    for (i = 0; i < n; i++)
        scanf("%d", &a[i]);

    mergesort(a, 0, n - 1);

    for (i = 0; i < n; i++)
        printf("%d ", a[i]);

    return 0;
}

6) Min-Max

#include <stdio.h>

int min, max;

void minmax(int a[], int l, int h) {
    int mid;

    if (l == h) {
        min = max = a[l];
        return;
    }

    if (h == l + 1) {
        if (a[l] < a[h]) {
            min = a[l];
            max = a[h];
        } else {
            min = a[h];
            max = a[l];
        }
        return;
    }

    mid = (l + h) / 2;

    int min1, max1, min2, max2;

    minmax(a, l, mid);
    min1 = min; max1 = max;

    minmax(a, mid + 1, h);
    min2 = min; max2 = max;

    min = (min1 < min2) ? min1 : min2;
    max = (max1 > max2) ? max1 : max2;
}

int main() {
    int n, a[100], i;

    scanf("%d", &n);

    for (i = 0; i < n; i++)
        scanf("%d", &a[i]);

    minmax(a, 0, n - 1);

    printf("Minimum: %d\n", min);
    printf("Maximum: %d", max);

    return 0;
}

7) OBST 

#include <stdio.h>
#include <limits.h>

int sum(int f[], int i, int j) {
    int s = 0;
    while (i <= j)
        s += f[i++];
    return s;
}

int obst(int f[], int n) {
    int cost[100][100], i, j, k, l;

    for (i = 0; i < n; i++)
        cost[i][i] = f[i];

    for (l = 2; l <= n; l++) {
        for (i = 0; i <= n - l; i++) {
            j = i + l - 1;
            cost[i][j] = INT_MAX;

            for (k = i; k <= j; k++) {
                int c = ((k > i) ? cost[i][k - 1] : 0) +
                        ((k < j) ? cost[k + 1][j] : 0) +
                        sum(f, i, j);

                if (c < cost[i][j])
                    cost[i][j] = c;
            }
        }
    }

    return cost[0][n - 1];
}

int main() {
    int n, key[100], freq[100], i;

    scanf("%d", &n);

    for (i = 0; i < n; i++)
        scanf("%d", &key[i]);

    for (i = 0; i < n; i++)
        scanf("%d", &freq[i]);

    printf("%d", obst(freq, n));

    return 0;
}
